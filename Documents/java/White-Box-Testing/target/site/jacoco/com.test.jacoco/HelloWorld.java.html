    public boolean isBirthday(int year, int month, int day) {
    	/**
    	 * TODO: You need to complete this method to determine whether a 
    	 * legitimate date of birth between 1990/01/01 and 2019/10/01 is 
    	 * formed or not when given the input year, month and day.
    	 */
        List<Integer> month30 = new ArrayList<Integer>(Arrays.asList(4, 6, 9, 11));
        List<Integer> month31 = new ArrayList<Integer>(Arrays.asList(1, 3, 5, 7, 8, 10 ,12));

        if(year < 1990 || year > 2019 || (year == 2019 && month > 10) || (year == 2019 && month == 10 && day > 1)){
            return false;
        }
        if(month < 1 || month > 12 || day < 1){
            return false;
        }
        if(month30.contains(month) && day > 30){
            return false;
        }
        if(month31.contains(month) && day > 31){
            return false;
        }
        if(year % 4 == 0 && month == 2 && day > 29){//闰年
            return false;
        }
        if(year % 4 != 0 && month == 2 && day > 28){//非闰年
            return false;
        }
    	return true;
    	
    }
